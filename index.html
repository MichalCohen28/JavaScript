מערכי אובייקטים

מערך הוא רשימה של ערכים. כשכל פריט ברשימה מכיל כמה שדות (כמו שם + שנים), הוא נקרא אובייקט. ביחד קוראים לזה מערך של אובייקטים.
JavaScript 
// מערך של כישורים — כל פריט הוא אובייקט עם שם ושנות ניסיון
skills_array = [
    {name: "Java",   years: 2},
    {name: "Python", years: 3},
    {name: "HTML",   years: 1}
];

// מערך של שפות — כל פריט הוא אובייקט עם שם ורמה
languages_array = [
    {name: "English", level: "Fluent"},
    {name: "Hebrew",  level: "Native"},
    {name: "Spanish", level: "Intermediate"}
];
💡 הסוגריים המסולסלים {} יוצרים אובייקט. הסוגריים המרובעים [] יוצרים מערך. כשמשלבים את שניהם — מקבלים מערך של אובייקטים.

בניית טבלה עם JS

הרעיון הוא לקחת מערך, לעבור על כל הפריטים שלו בלולאה, ולבנות HTML של טבלה שיוצג בדפדפן.
1
מוצאים את הטבלה בדף בעזרת document.querySelector
2
מגדירים כותרות עם שורת <tr><th>
3
לולאה על המערך — בכל סיבוב מוסיפים שורה עם הנתונים
4
שמים הכל בתוך הטבלה בעזרת innerHTML
JavaScript 
function show_skills_table() {

    // שלב 1 — מוצאים את הטבלה בדף
    let table = document.querySelector("#Skills-table");

    // שלב 2 — כותרות
    let txt = "<tr><th>Skill</th><th>Years</th></tr>";

    // שלב 3 — לולאה על המערך
    for (let i = 0; i < skills_array.length; i++) {
        txt += `<tr>
            <td>${skills_array[i].name}</td>
            <td>${skills_array[i].years}</td>
        </tr>`;
    }

    // שלב 4 — מכניסים לטבלה
    table.innerHTML = txt;
}
🏷️ <tr> = Table Row (שורה חדשה)  |  <td> = Table Data (תא חדש)  |  <th> = Table Header (כותרת)

כפתור מיון טבלה

הפונקציה sort() מסדרת מערך. כדי שהיא תדע מה "לפני" ומה "אחרי", אנחנו נותנים לה פונקציית השוואה שמחזירה -1, 0, או 1.
-1

a בא לפני b
(a קטן יותר או עדיף)

1

a בא אחרי b
(b קטן יותר או עדיף)

0

הם שווים
אין שינוי בסדר

JavaScript 
// מחכה ללחיצה על הכפתור
document.querySelector("#btn_sort_by_year").onclick = () => {

    skills_array.sort((a, b) => {

        if (a.years > b.years)
            return -1;  // a לפני b

        else if (a.years < b.years)
            return 1;   // b לפני a

        return 0;        // שווים — ללא שינוי
    });

    show_skills_table(); // רענון תצוגה
};
ערך מוחזר מה ה-JS עושה תוצאה
-1 הזזת a לפני b מיון יורד (הגדול קודם)
1 הזזת b לפני a מיון עולה (הקטן קודם)
0 ללא שינוי הסדר נשמר

הוספת נתונים לטבלה

כדי לאפשר למשתמש להוסיף שורות לטבלה, ב-HTML מוסיפים שדות קלט (input), וב-JS כותבים פונקציה שקוראת את הנתונים, דוחפת אותם למערך בעזרת push(), ואז מרענת את הטבלה.
HTML 
<!-- שדות קלט + כפתור -->
<input type="text" id="name-input"   placeholder="Name"  />
<input type="text" id="course-input" placeholder="Course"/>
<input type="text" id="grade-input"  placeholder="Grade" />

<button onclick="addStudent()">Add Student</button>
JavaScript 
function addStudent() {

    // שלב 1 — קוראים את הערכים מהשדות
    let name   = document.querySelector("#name-input").value;
    let course = document.querySelector("#course-input").value;
    let grade  = document.querySelector("#grade-input").value;

    // שלב 2 — דוחפים אובייקט חדש למערך
    // ה-+ לפני grade הופך string למספר
    Students.push({name: name, course: course, grade: +grade});

    // שלב 3 — מאפסים את שדות הקלט
    document.querySelector("#add-student-form").reset();

    // שלב 4 — מרעננים את הטבלה
    showTable();
}
⚡ שים לב ל-+grade — הפלוס הופך את הטקסט שהמשתמש הקליד (שהוא תמיד string) למספר אמיתי, כדי שנוכל לעשות עליו חשבון ומיון.
